# Install Odoo via the `.deb` package

## 1. بروزرسانی سیستم
sudo apt update && sudo apt upgrade -y

## 2.📦 نصب پیش‌نیازها
sudo apt install python3-pip build-essential python3-dev libpq-dev \
libxml2-dev libxslt1-dev zlib1g-dev libsasl2-dev libldap2-dev \
libjpeg-dev libffi-dev libtiff5-dev libopenjp2-7 liblcms2-dev \
libwebp-dev libharfbuzz-dev libfribidi-dev libxcb1-dev nodejs npm -y


## 3. 🐘 نصب پایگاه داده PostgreSQL
sudo apt install postgresql postgresql-contrib -y
sudo -u postgres createuser -s odoo

## 4. 📄 نصب wkhtmltopdf برای تولید PDF
sudo apt install wkhtmltopdf -y

## 5. 📥 نصب فایل `.deb` اودو

اگر فایل `odoo_18.0+e.latest_all_odoers_ir.deb` را در پوشه Downloads دارید:
cd ~/Downloads
sudo dpkg -i odoo_18.0+e.latest_all_odoers_ir.deb
sudo apt --fix-broken install -y
